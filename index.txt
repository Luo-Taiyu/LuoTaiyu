        // 创建水墨斑点效果
        function createInkBlots() {
            const colors = ['rgba(42, 92, 170, 0.05)', 'rgba(201, 58, 58, 0.05)', 'rgba(232, 185, 35, 0.05)'];
            const body = document.body;
            
            for (let i = 0; i < 5; i++) {
                const inkBlot = document.createElement('div');
                inkBlot.className = 'ink-blot';
                
                // 随机大小和位置
                const size = Math.random() * 300 + 100;
                const left = Math.random() * 100;
                const top = Math.random() * 100;
                const color = colors[Math.floor(Math.random() * colors.length)];
                
                inkBlot.style.width = `${size}px`;
                inkBlot.style.height = `${size}px`;
                inkBlot.style.left = `${left}%`;
                inkBlot.style.top = `${top}%`;
                inkBlot.style.background = `radial-gradient(circle, ${color} 0%, transparent 70%)`;
                
                body.appendChild(inkBlot);
            }
        }

        // 初始化水墨斑点
        createInkBlots();

        // 鼠标跟随效果 - 简约水墨点
        document.addEventListener('mousemove', function(e) {
            // 限制创建频率
            if (Math.random() > 0.7) {
                const dot = document.createElement('div');
                dot.className = 'ink-blot';
                
                const size = Math.random() * 20 + 5;
                const color = `rgba(${Math.random() > 0.5 ? '42, 92, 170' : '201, 58, 58'}, ${Math.random() * 0.1 + 0.05})`;
                
                dot.style.width = `${size}px`;
                dot.style.height = `${size}px`;
                dot.style.left = `${e.pageX - size/2}px`;
                dot.style.top = `${e.pageY - size/2}px`;
                dot.style.background = `radial-gradient(circle, ${color} 0%, transparent 70%)`;
                dot.style.transition = 'opacity 1s ease';
                
                document.body.appendChild(dot);
                
                // 逐渐消失并移除
                setTimeout(() => {
                    dot.style.opacity = '0';
                    setTimeout(() => {
                        if (dot.parentNode) {
                            dot.parentNode.removeChild(dot);
                        }
                    }, 1000);
                }, 100);
            }
        });

        // 滚动动画效果
        function checkScroll() {

                    element.style.transform = 'translateY(0)';
